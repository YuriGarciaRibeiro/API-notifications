# API Access Segregation (UI + Integration) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement route and identity segregation so UI users use JWT under `/api/ui/*`, internal systems send notifications via API Key under `/api/integration/*`, and UI visibility is filtered by source-system access.

**Architecture:** Keep the existing Clean Architecture and Minimal API style. Add a small integration-identity slice (`IntegrationClient` + `UserSourceAccess`), add API Key authentication for `/api/integration`, and split notification endpoints into UI and Integration groups while reusing existing notification use cases. Enforce source-based visibility in query handlers using current user context plus DB access mappings.

**Tech Stack:** .NET 10, ASP.NET Core Minimal APIs, MediatR, FluentResults, EF Core (Npgsql), xUnit + WebApplicationFactory.

```csharp
using System.Security.Claims;
using System.Text.Encodings.Web;
using Microsoft.AspNetCore.Authentication;
using Microsoft.Extensions.Options;
using NotificationSystem.Application.Interfaces;

namespace NotificationSystem.Api.Authentication;

public class ApiKeyAuthenticationHandler(
    IOptionsMonitor<AuthenticationSchemeOptions> options,
    ILoggerFactory logger,
    UrlEncoder encoder,
    IIntegrationClientRepository integrationClientRepository,
    IApiKeyHasher apiKeyHasher)
    : AuthenticationHandler<AuthenticationSchemeOptions>(options, logger, encoder)
{
    public const string SchemeName = "ApiKey";

    protected override async Task<AuthenticateResult> HandleAuthenticateAsync()
    {
        if (!Request.Headers.TryGetValue("X-API-Key", out var apiKeyHeader) || string.IsNullOrWhiteSpace(apiKeyHeader))
            return AuthenticateResult.Fail("Missing API key");

        var hash = apiKeyHasher.Hash(apiKeyHeader.ToString());
        var client = await integrationClientRepository.GetActiveByApiKeyHashAsync(hash, Context.RequestAborted);

        if (client is null)
            return AuthenticateResult.Fail("Invalid API key");

        var claims = new[]
        {
            new Claim("integration_client_id", client.Id.ToString()),
            new Claim("source_system_id", client.Id.ToString()),
            new Claim("auth_type", "integration")
        };

        var identity = new ClaimsIdentity(claims, SchemeName);
        var principal = new ClaimsPrincipal(identity);
        var ticket = new AuthenticationTicket(principal, SchemeName);

        return AuthenticateResult.Success(ticket);
    }
}
```

// src/NotificationSystem.Api/Endpoints/IntegrationNotificationEndpoints.cs
public static class IntegrationNotificationEndpoints
{
    public static IEndpointRouteBuilder MapIntegrationNotificationEndpoints(this IEndpointRouteBuilder endpoints)
    {
        var group = endpoints.MapGroup("/api/integration/notifications")
            .WithTags("Integration Notifications")
            .RequireAuthorization();

        group.MapPost("/", async (
            [FromBody] CreateNotificationCommand command,
            HttpContext httpContext,
            IMediator mediator,
            CancellationToken cancellationToken) =>
        {
            var sourceSystemClaim = httpContext.User.FindFirst("source_system_id")?.Value;
            if (!Guid.TryParse(sourceSystemClaim, out var sourceSystemId))
                return Results.Unauthorized();

            var integrationCommand = command with
            {
                Origin = NotificationOrigin.Api,
                SourceSystemId = sourceSystemId
            };

            var result = await mediator.Send(integrationCommand, cancellationToken);
            return result.ToIResult(201);
        });

        return endpoints;
    }
}
